function loadSocial(){ if( window.matchMedia('(min-width: 950px)').matches ){ return false; } var flag = false; if(window.VK){ VK.Widgets.Like("vk_like", {type: "mini"}); $('div[name=vkSocial]').removeClass('frOff'); flag = true; } scripts = { 'yaSocial' : '//yandex.st/share/share.js', 'gSocial' : 'https://apis.google.com/js/plusone.js' } $.each(scripts, function(key, val){ var script = document.createElement('script'); script.src = val; script.onload = function(){ $('div[name='+key+']').removeClass('frOff'); flag = true; } document.body.appendChild(script); }); if(flag){ $('div[name=social]').removeClass('frOff'); } return flag; }
